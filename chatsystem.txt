    username varchar(255) NOT NULL,

    age int,

(

    groupname varchar(255) NOT NULL,
    grouppermission varchar(255),
    admins int NOT NULL,

    membercount int,
    chatlock varchar(255),

create table chatsystem.messages
(
    messageid int primary key unique,
    receiver int NOT NULL,
    sender int NOT NULL,
    readstatus varchar(255) DEFAULT 'unread',
    descriptions varchar(255),
    timestamps varchar(255),
    attachmentid int,
    imagetypes varchar(255),
    foreign key(receiver) references users(userid),
    foreign key(sender) references users(userid)
);

create table chatsystem.chat
(
    chatid int primary key unique,
    participantname varchar(255) NOT NULL,
    disappearingmessages varchar(255),
    calls varchar(255)
);

create table chatsystem.notifications
(
    notificationid int primary key unique,
    userid int NOT NULL,
    notificationtype varchar(255) NOT NULL,
    content varchar(255),
    timestamps TIMESTAMP DEFAULT CURRENT_TIMESTAMP,
    foreign key(userid) references users(userid)
);

create table chatsystem.settings
(
    settingsid int primary key unique,
    chathistory varchar(255),
    storageanddata varchar(255),
    notificationsettings varchar(255),
    blockedcontacts varchar(255),
    theme varchar(255),
    userid int NOT NULL,
    helpcenter varchar(255),
    accountinformation varchar(255),
    foreign key(userid) references users(userid)
);

create table chatsystem.attachment
(
    attachmentid int primary key unique,
    filename varchar(255),
    filesize varchar(255),
    content varchar(255),
    messageid int,
    filetype varchar(255),
    foreign key(messageid) references messages(messageid)
);

SELECT u.userid, u.username, g.groupname, g.grouppermission, g.creationdate
FROM chatsystem.users u
INNER JOIN chatsystem.groupchat g 
ON u.userid = g.admins
WHERE g.grouppermission = 'private'
GROUP BY u.userid, u.username, g.groupname, g.grouppermission, g.creationdate;
